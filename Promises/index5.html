conset exchangeRates = { 'USD' : 1.0, 'EUR' : 0.85, 'GBP' : 0.75, }; function convertCurrency(amount, sourceCurrency, targetCurrency){ return new Promise((resolve, reject) => { if(exchangeRates[sourceCurrency] && exchangeRates[targetCurrency]){ let result = (amount * (exchangeRates[targetCurrency] / exchangeRates[sourceCurrency])).toFixed(2); resolve(result); } else{ reject("Error converting currency : Invalid Currency"); } }) }
